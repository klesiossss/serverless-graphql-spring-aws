## Basic Commands
# user
query User {
user(id: "79da92de-c7e4-4481-8964-dc520f76842f") {
id
name
email
}
}

# users

query Users {
users {
id
name
email
}
}


# create
mutation CreateUser {
createUser(input: {name: "pedro", email: "pedro@gmail.com"}) {
name
email
}
}


# update
mutation UpdateUser {
updateUser(input: {name: "antony swants", email: "antony@gmail.com"}, id: "79da92de-c7e4-4481-8964-dc520f76842f") {
id
name
email
}
}

# delete
mutation DeleteUser {
deleteUser(id: "1") {
id
name
email
}
}
